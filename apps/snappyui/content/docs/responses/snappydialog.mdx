## Variations

### Form Dialog

<Tabs items={["Preview", "Code"]}>
  <Tab value="Preview">
    <div className="flex flex-col items-center justify-center mb-6 rounded-lg py-5 px-5">
      <DialogDemoTwo />
    </div>
  </Tab>
  <Tab value="Code">
    ```tsx
    import {
      Dialog,
      DialogActions,
      DialogButton,
      DialogForm,
      DialogField,
      DialogLabel,
      DialogInput
    } from "@/components/ui/snappy-dialog";

    const [isFormOpen, setIsFormOpen] = useState(false);
    const [formData, setFormData] = useState({ name: '', email: '' });

    const handleSubmit = (e: React.FormEvent) => {
      e.preventDefault();
      console.log(`Form submitted with: ${formData.name}, ${formData.email}`);
    };

    <div className="p-4 rounded-md">
      <button
        className="px-6 py-2 bg-background border-1 text-dark dark:text-white rounded-md"
        onClick={() => setIsFormOpen(true)}
      >
        Open
      </button>

      <Dialog
        isOpen={isFormOpen}
        onClose={() => setIsFormOpen(false)}
        title="Form Dialog"
      >
        <DialogForm onSubmit={handleSubmit}>
          <DialogField>
            <DialogLabel htmlFor="name">Name</DialogLabel>
            <DialogInput
              id="name"
              value={formData.name}
              onChange={(e) => setFormData({ ...formData, name: e.target.value })}
              required
            />
          </DialogField>

          <DialogField>
            <DialogLabel htmlFor="email">Email</DialogLabel>
            <DialogInput
              id="email"
              type="email"
              value={formData.email}
              onChange={(e) => setFormData({ ...formData, email: e.target.value })}
              required
            />
          </DialogField>

          <DialogActions>
            <DialogButton type="button" onClick={() => setIsFormOpen(false)}>Cancel</DialogButton>
            <DialogButton type="submit" variant="primary">Submit</DialogButton>
          </DialogActions>
        </DialogForm>
      </Dialog>
    </div>
    ```
  </Tab>
</Tabs>

# Dialog Component Props

### Dialog

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | Required | Controls the visibility of the dialog |
| `onClose` | `() => void` | Required | Function called when the dialog should close |
| `title` | `string` | `undefined` | Optional title for the dialog |
| `showCloseButton` | `boolean` | `true` | Whether to show the close button (X) |
| `children` | `React.ReactNode` | Required | Content of the dialog |
| `className` | `string` | `undefined` | Additional CSS classes |
| `variant` | `'default' \| 'dark' \| 'light'` | `'default'` | Controls the color scheme |
| `position` | `'center' \| 'top'` | `'center'` | Controls the position of the dialog |
| `...props` | `React.HTMLAttributes<HTMLDivElement>` | - | Any other HTML div props |

### DialogForm

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `onSubmit` | `(e: React.FormEvent) => void` | Required | Form submission handler |
| `children` | `React.ReactNode` | Required | Form content |
| `className` | `string` | `undefined` | Additional CSS classes |

### DialogContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `React.ReactNode` | Required | Content |
| `className` | `string` | `undefined` | Additional CSS classes |
| `...props` | `React.HTMLAttributes<HTMLDivElement>` | - | Any other HTML div props |

### DialogActions

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `React.ReactNode` | Required | Action buttons or content |
| `className` | `string` | `undefined` | Additional CSS classes |
| `...props` | `React.HTMLAttributes<HTMLDivElement>` | - | Any other HTML div props |

### DialogField

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `React.ReactNode` | Required | Field content |
| `className` | `string` | `undefined` | Additional CSS classes |
| `...props` | `React.HTMLAttributes<HTMLDivElement>` | - | Any other HTML div props |

### DialogLabel

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `htmlFor` | `string` | Required | ID of the form control this label is for |
| `children` | `React.ReactNode` | Required | Label text |
| `className` | `string` | `undefined` | Additional CSS classes |
| `...props` | `React.LabelHTMLAttributes<HTMLLabelElement>` | - | Any other HTML label props |

### DialogInput

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `className` | `string` | `undefined` | Additional CSS classes |
| `...props` | `React.InputHTMLAttributes<HTMLInputElement>` | - | Any HTML input props |

### DialogButton

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'primary' \| 'secondary'` | `'secondary'` | Button style variant |
| `children` | `React.ReactNode` | Required | Button text/content |
| `className` | `string` | `undefined` | Additional CSS classes |
| `...props` | `React.ButtonHTMLAttributes<HTMLButtonElement>` | - | Any HTML button props |
