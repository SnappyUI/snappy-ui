
### Props

| Prop      | Type                              | Default         | Description                                        |
| --------- | --------------------------------- | --------------- | -------------------------------------------------- |
| label     | `string`                          | `undefined`     | Text label for the checkbox                        |
| variant   | `CheckboxVariant`                 | `bounce`        | Animation style to apply                           |
| checked   | `boolean`                         | `false`         | Controlled checked state                           |
| onChange  | `(checked: boolean) => void`      | `undefined`     | Callback when checked state changes                |
| disabled  | `boolean`                         | `false`         | Whether the checkbox is disabled                   |
| className | `string`                          | `''`            | Additional CSS classes                             |
| id        | `string`                          | `undefined`     | HTML ID attribute (auto-generated if not provided) |


### Form Integration

```tsx
import Check from "@/components/ui/snappy-checkbox";
import { useState } from 'react';
import { cn } from '@/lib/utils';

export default function ContactForm() {
  const [formState, setFormState] = useState({
    name: '',
    email: '',
    acceptTerms: false,
    acceptMarketing: false
  });
  
  const handleSubmit = (e) => {
    e.preventDefault();
    // Submit form logic
    console.log('Form submitted:', formState);
  };
  
  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      {/* Other form fields */}
      
      <div className="space-y-2">
        <Check
          label="I accept the terms and conditions"
          variant="bounce"
          checked={formState.acceptTerms}
          onChange={(checked) => setFormState({...formState, acceptTerms: checked})}
          className={cn(
            formState.acceptTerms ? 'text-green-700' : 'text-gray-700'
          )}
        />
        
        <Check
          label="Send me marketing emails"
          variant="fade"
          checked={formState.acceptMarketing}
          onChange={(checked) => setFormState({...formState, acceptMarketing: checked})}
        />
      </div>
      
      <button 
        type="submit"
        className="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700"
        disabled={!formState.acceptTerms}
      >
        Submit
      </button>
    </form>
  );
}
```
