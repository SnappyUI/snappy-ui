## Basic Usage

<Tabs items={["Preview", "Code"]}>
  <Tab value="Preview">
    <div className="flex flex-col items-center justify-center mb-6 rounded-lg py-5 px-5">
      <SnappySignIn />
    </div>
  </Tab>

  <Tab value="Code">
    ```tsx
    import {
      LoginPage,
      LoginHeader,
      EmailInput,
      PasswordInput,
      ForgotPasswordLink,
      SubmitButton,
      SignUpLink,
    } from "@/components/ui/snappy-login";

    <LoginPage>
      <LoginHeader title="Welcome Back" subtitle="Log in to your account" />
      <EmailInput />
      <PasswordInput />
      <ForgotPasswordLink />
      <SubmitButton
        onClick={(data) => {
          return new Promise((resolve) => {
            setTimeout(() => {
              console.log("I will get this data while submitting", data);
              resolve(true); // This is important
            }, 5000);
          });
        }}
      />
      <SignUpLink />
    </LoginPage>
    ```
  </Tab>
</Tabs>

## Props

### LoginPage

| Name       | Type        | Default     | Description                |
| ---------- | ----------- | ----------- | -------------------------- |
| children   | `ReactNode` | `-`         | Contents of the component  |
| className  | `string`    | `-`         | Tailwind utility classes   |

### LoginHeader

| Name       | Type        | Default                                           | Description                       |
| ---------- | ----------- | ------------------------------------------------- | --------------------------------- |
| title      | `string`    | `"Sign In"`                                       | Main heading text                 |
| subtitle   | `string`    | `"Enter your credentials to access your account"` | Subheading text                   |
| logo       | `ReactNode` | `null`                                            | Optional logo element             |
| children   | `ReactNode` | `-`                                               | Additional elements inside header |
| className  | `string`    | `-`                                               | Tailwind utility classes          |

### EmailInput

| Name        | Type     | Default             | Description              |
| ----------- | -------- | ------------------- | ------------------------ |
| label       | `string` | `"Email"`           | Label text               |
| placeholder | `string` | `"you@example.com"` | Input placeholder        |
| className   | `string` | `-`                 | Tailwind utility classes |

### PasswordInput

| Name       | Type      | Default       | Description               |
| ---------- | --------- | ------------- | ------------------------- |
| label      | `string`  | `"Password"`  | Label text                |
| placeholder| `string`  | `"••••••••"`  | Input placeholder         |
| showToggle | `boolean` | `true`        | Show/hide password toggle |
| className  | `string`  | `-`           | Tailwind utility classes  |

### ForgotPasswordLink

| Name       | Type     | Default              | Description              |
| ---------- | -------- | -------------------- | ------------------------ |
| text       | `string` | `"Forgot password?"` | Link text                |
| href       | `string` | `"#"`                | Link destination         |
| className  | `string` | `-`                  | Tailwind utility classes |

### SubmitButton

| Name        | Type     | Default           | Description                    |
| ----------- | -------- | ----------------- | ------------------------------ |
| text        | `string` | `"Sign In"`       | Button label                   |
| loadingText | `string` | `"Signing in..."` | Text shown when loading        |
| className   | `string` | `-`               | Tailwind utility classes       |
| onClick     | `cb`     | `-`               | Optional custom submit handler |

### SignUpLink

| Name      | Type     | Default                            | Description              |
| --------- | -------- | ---------------------------------- | ------------------------ |
| text      | `string` | `"Don't have an account? Sign up"` | Link text                |
| href      | `string` | `"#"`                              | Link destination         |
| className | `string` | `-`                                | Tailwind utility classes |
